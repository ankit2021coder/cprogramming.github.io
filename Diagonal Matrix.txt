
#include <stdio.h>

int main()
{
   int x[3][3];
   int i,j;
   printf("Enter a matrix :");
   for(i=0;i<3;i++)
   {
       for(j=0;j<3;j++)
       {
           scanf("%d",&x[i][j]);
       }
   }
    for(i=0;i<3;i++)
   {
       for(j=0;j<3;j++)
       {
           if(i==j)
           {
               printf("%d",x[i][j]);
           }
       }
   }
}
